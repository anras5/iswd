import random

def constructParents(N):
    def tournament(idxs):
        return min(random.sample(idxs, 2))

    def get_parents(N):
        available = list(range(N))
        first = tournament(available)
        available.remove(first)
        second = tournament(available)
        return [first, second]
        
    return [get_parents(N) for _ in range(0, N)]

def mutate(x, prob, std):
    for i in range(len(x)):
        if np.random.rand() < prob:
            if i in [2, 3, 4]:
                x[i] = 1 - x[i]
            else:
                x[i] = x[i] + np.random.normal(0, std)
                x[i] = max(0, min(1, x[i]))

def constructOffspring(P, parents, gen, std):
    O = []
    prob = 1.0 / 5.0
    for i in range(len(parents)):
        xO = getCrossed(P[parents[i][0]].x, P[parents[i][1]].x)
        mutate(xO, prob, std)
        O.append(Solution(xO, evaluate(xO), str(gen) + "-" + str(i)))
    return O

# Pomocnicza metoda okrelajca zajcie relacji dominacji
def dominates(sj, sk):
    if sj.f[0] < sk.f[0] and sj.f[1] <= sk.f[1]: return True
    if sj.f[0] <= sk.f[0] and sj.f[1] < sk.f[1]: return True
    return False

### TODO Ponisza funkcja winna realizowa przydzia rozwiza do frontw niezdominowanych
### Wyjciem powinna by lista list indeksw rozwiza w P, przydzielonych do odpowiednich frontw.
### Np. wyjcie: [[3,4,0],[1,5],[2]] oznacza, e rozwizania nr 0, 3, 4 w P s w pierszym froncie,
### 1 i 5 w drugim, a rozwizanie numer 2 w ostatnim. 

def getNonDominatedFronts(P):
    def getFront(P):
        S = sorted(P, key=lambda x: (x.f[0], x.f[1]))
        R = []
        if len(S) == 1:
            return S
        
        T = getFront(S[:len(S)//2])
        B = getFront(S[len(S)//2:])
        N = [b for b in B if not any([dominates(t, b) for t in T])]
        return T + N

    fronts_idx = []
    X = P.copy()
    while len(X) > 0:
        front = getFront(X)
        fronts_idx.append([P.index(f) for f in front])
        X = [p for p in X if p not in front]
    
    return fronts_idx

### TODO ponisza metoda winna oblicza wartoci crowding distance (cd)
# dla rozwiaza w P. UWAGA: cd jest obliczany dla niezdominowanych frontw z osobna;
# obliczajc cd dla rozwiza w jednym froncie "zapominamy" o pozostaych rozwizaniach.
# Metoda winna zwrci wektor uzyskanych cd

def getCrowdingDistances(F, P):

    result = [0 for _ in P]
    for front in F:
        X = [(i, P[i]) for i in front]  # (indeks w P, rozwizanie)
        for m in range(2):
            X = sorted(X, key=lambda x: x[1].f[m])
            for i in range(len(X)):
                if i in [0, len(X)-1] or abs(X[-1][1].f[m]-X[0][1].f[m]) < 1e-5:
                    result[X[i][0]] = np.inf
                else:
                    result[X[i][0]] = result[X[i][0]] + (X[i+1][1].f[m] - X[i-1][1].f[m])/(X[-1][1].f[m]-X[0][1].f[m])
                
    return result

### Ponisza metoda przydziela rozwizaniom ich numery niezdominowanych frontw i wartoci cd,
# a take sortuje rozwizania w populacji w oparciu o te wartoci. 
# W przypadku, gdy useCD = False, miara cd nie jest uwzgldniona przy sortowaniu.
# Przetestowanie metody dla useCD = True i False pozwoli zaobserwowa zysk wynikajcy
# z wykorzystania miary cd.

def applyScoresAndSort(P, useCD = True):
    F = getNonDominatedFronts(P)
    CD = [0 for _ in P]
    if useCD:
        CD = getCrowdingDistances(F, P)

    for s, f in enumerate(F):
         for i in f: 
            P[i].front = s
            P[i].cd = CD[i]
   
    P.sort(key=lambda x: x.cd, reverse=True)
    P.sort(key=lambda x: x.front) # Kryterium sortowania

# TODO Ponisza funkcja winna oblicza pomocnicze statystyki, ktre pomog
# uzasadni wykorzystanie miary cd w obliczeniach.
# Metoda tak winna zwrci redni, maksymaln, minimaln warto 
# cd dla rozwiza, oraz odchylenie standardowe tych wartoci
# w populacji P. Przy obliczaniu statystyk naley pomin wartoci graniczne (nieskoczono),
# uzyskiwane przez wartoci graniczne
# Jak naley interpretowa te miary?

def getStatsCD(P):
    to_stats = [p.cd for p in P if not np.isinf(p.cd)]
    if len(to_stats) > 0:
        return np.mean(to_stats), np.max(to_stats), np.min(to_stats), np.std(to_stats)
    else:
        return 0, 0, 0, 0

Odpowied

Lepsze dziaanie mona zaobserwowa dla algorytmu wykorzystujcego miar Crowd Distance, poniewa otrzymany wynik w lepszym stopniu modeluje cay front Pareto, a nie jedynie wybrane odcinki z frontu.

Dla chtnych: algorytmy ewolucyjne s losowe. Z tego powodu jednokrotne ich uruchomienie nie jest wiarygodne. By uzyska bardziej wiarygodny wynik, algorytm moe zosta uruchomiony np. 20 razy a uzyskane statystyki urednione. Dodatkowo mona je obliczy dla kadej generacji (nie tylko w ostatniej) i wykreli ich zbieno na wykresie liniowym. Na jednym wykresie mona wykreli wyniki uzyskane przez oba warianty algorytmu - dziki czemu rnice powinny byc lepiej widoczne. Prosz zauway jednak, e statystki dla cd nie s moliwie najlepsz miar oceny jakoci dziaania algorytmu. Na samym pocztku, gdy rozwizania s silnie rozproszone, ich wartoci cd powinny by bardzo wysokie, co jest w sprzecznoci z zaoeniem, e "im wiksze cd tym lepiej". Aby wiarygodniej ocenia takie algorytmy, naley take wykorzystywa miary, ktre szacuj odlego rozwiza do frontu Pareta. Mona zaproponowa tak miar i wykorzysta j do pokazania "bliskoci" populacji do tego frontu. Dopiero jednoczesne zestawienie takiej miary i miary oceniajcej rozproszenie rozwiza umoliwi lepsze porwnanie algorytmw.
